229#define ATT_FIND_INFO_REQ_FIXED_SIZE 4

236#define ATT_FIND_BY_TYPE_VALUE_REQ_FIXED_SIZE 6

242
243// Length of Read By Type Response's fixed fields: Length (1)

245#define ATT_READ_BY_TYPE_RSP_HDR_SIZE ( ATT_OPCODE_SIZE + ATT_READ_BY_TYPE_RSP_FIXED_SIZE )

296// Handle and 16-bit Bluetooth UUID pair indexes
297#define ATT_BT_PAIR_HANDLE_IDX( i ) ( (i) * (2 + ATT_BT_UUID_SIZE) )
298#define ATT_BT_PAIR_UUID_IDX( i ) ( ATT_BT_PAIR_HANDLE_IDX( (i) ) + 2 )
299
300#define ATT_BT_PAIR_HANDLE( info, i ) ( BUILD_UINT16( (info)[ATT_BT_PAIR_HANDLE_IDX((i))], \
301 (info)[ATT_BT_PAIR_HANDLE_IDX((i))+1] ) )
302#define ATT_BT_PAIR_UUID( info, i ) ( BUILD_UINT16( (info)[ATT_BT_PAIR_UUID_IDX((i))], \
303 (info)[ATT_BT_PAIR_UUID_IDX((i))+1] ) )
304
305// Handle and 128-bit UUID pair indexes
306#define ATT_PAIR_HANDLE_IDX( i ) ( (i) * (2 + ATT_UUID_SIZE) )
307#define ATT_PAIR_UUID_IDX( i ) ( ATT_PAIR_HANDLE_IDX( (i) ) + 2 )
308
309#define ATT_PAIR_HANDLE( info, i ) ( BUILD_UINT16( (info)[ATT_PAIR_HANDLE_IDX((i))], \
310 (info)[ATT_PAIR_HANDLE_IDX((i))+1] ) )
311
312/*********************************************************************
313 * ATT Find By Type Value Response macros
314 */
315// Attribute Handle and Group End Handle pair indexes
316#define ATT_ATTR_HANDLE_IDX( i ) ( (i) * (2 + 2) )
317#define ATT_GRP_END_HANDLE_IDX( i ) ( ATT_ATTR_HANDLE_IDX( (i) ) + 2 )
318
319#define ATT_ATTR_HANDLE( info, i ) ( BUILD_UINT16( (info)[ATT_ATTR_HANDLE_IDX((i))], \
320 (info)[ATT_ATTR_HANDLE_IDX((i))+1] ) )
321#define ATT_GRP_END_HANDLE( info, i ) ( BUILD_UINT16( (info)[ATT_GRP_END_HANDLE_IDX((i))], \
322 (info)[ATT_GRP_END_HANDLE_IDX((i))+1] ) )
323
324/*********************************************************************
325 * ATT Read Multiple Request macros
326 */
327// Attribute Handle index
328#define ATT_HANDLE_IDX( i ) ( (i) * 2 )
329
330#define ATT_HANDLE( handles, i ) ( BUILD_UINT16( (handles)[ATT_HANDLE_IDX((i))], \
331 (handles)[ATT_HANDLE_IDX((i))+1] ) )
