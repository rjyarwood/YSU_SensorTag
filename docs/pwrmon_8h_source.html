136// voltage register unit description @anchor PWRMON_REGISTER ##
137// Voltage thresholds are defined in "units" compatible with the CC26xx
138// AON_BATMON_O_BAT register (see Technical Reference Manual for details).
139// The register provides the voltage reading in 11 bits, masked by 0x7FF,
140// where 'whole' volts are located in bits 8-10 (0x700) and the 'fraction'
141// is located in bits 0-7 (0x0FF). The 'fraction' (0.00-0.99) is scaled
142// linearly across the digital values of 0x00-0xFF, with 0x80 = 0.50 volt,
143// 0xFF = 0.99 volt, etc.
144// The default values provided below indicate settings slightly above and
145// somewhat below the CC26xx brown-out threshold. It is intended that the
146// user will modify these values appropriately for their system needs.
147
148#if !defined (MIN_VDD_INIT)
149#define MIN_VDD_INIT 0x1CE // 1.80 volts (0.80=206/256 -> 206=0xCE)
150#endif
151
152#if !defined (MIN_VDD_POLL)
153#define MIN_VDD_POLL 0x1DA // 1.85 volts (0.85=218/256 -> 218=0xDA)
154#endif
155
156#if !defined (MIN_VDD_FLASH)
157#define MIN_VDD_FLASH 0x180 // 1.50 volts (0.50=128/256 -> 128=0x80)
158#endif
159
160// This macro converts a register value into millivolts.
161#define PWRMON_REG_TO_MV(reg) ((((reg & 0xFFFF) >> 8) * 1000) + \
162 (((reg & 0xFF)*1000)>>8));
163
164// This macro converts millivolts into a register value.
165#define PWRMON_MV_TO_REG(volt) (((volt / 1000) << 8) + \
166 (((volt * 256)/1000) & 0xFF));

194/*******************************************************************************
195 * @fn PWRMON_setMonitor()
196 *
197 * @brief set the monitoring rate of the voltage. when voltage is detected to
198 * be lower than the threasholod set by PWRMON_setThreshold, the
199 * callback is called.
200 * the threshold and the callback needs to be set first by calling the
201 * API PWRMON_setThreshold.
202 *
203 * @param[in] mode: the monitoring mode:
204 * - PWRMON_MODE_LOW, CB is called when voltage is lower
205 * than threshold
206 * - PWRMON_MODE_HIGH, CB is called when voltage is higher
207 * than threshold
208 * - PWRMON_MODE_ALWAYS, CB is always called.
209 * @param[in] threshold: threshold in mV
210 * @param[in] period: in mSec between each measurement (min 10, max 65535)
211 * @param[in] cBack: callback to call when voltage goes under the
212 * threshold
213 *
214 * @return None
215 *******************************************************************************
216 */
