85// Write the 32bit value of 'val' in little endian format to the buffer pointed
86// to by pBuf, and increment pBuf by 4
87#define UINT32_TO_BUF_LITTLE_ENDIAN(pBuf,val) \
88 do { \
89 *(pBuf)++ = ((((uint32)(val)) >> 0) & 0xFF); \
90 *(pBuf)++ = ((((uint32)(val)) >> 8) & 0xFF); \
91 *(pBuf)++ = ((((uint32)(val)) >> 16) & 0xFF); \
92 *(pBuf)++ = ((((uint32)(val)) >> 24) & 0xFF); \
93 } while (0)
94
95// Return the 32bit little-endian formatted value pointed to by pBuf, and increment pBuf by 4
96#define BUF_TO_UINT32_LITTLE_ENDIAN(pBuf) (((pBuf) += 4), BUILD_UINT32((pBuf)[-4], (pBuf)[-3], (pBuf)[-2], (pBuf)[-1]))

109 * This macro is for use by other macros to form a fully valid C statement.
110 * Without this, the if/else conditionals could show unexpected behavior.
111 *
112 * For example, use...
113 * #define SET_REGS() st( ioreg1 = 0; ioreg2 = 0; )
114 * instead of ...
115 * #define SET_REGS() { ioreg1 = 0; ioreg2 = 0; }
116 * or
117 * #define SET_REGS() ioreg1 = 0; ioreg2 = 0;
118 * The last macro would not behave as expected in the if/else construct.
119 * The second to last macro will cause a compiler error in certain uses
120 * of if/else construct
121 *
122 * It is not necessary, or recommended, to use this macro where there is
123 * already a valid C statement. For example, the following is redundant...
124 * #define CALL_FUNC() st( func(); )
125 * This should simply be...
126 * #define CALL_FUNC() func()
127 *
128 * (The while condition below evaluates false without generating a
129 * constant-controlling-loop type of warning on most compilers.)
130 */
